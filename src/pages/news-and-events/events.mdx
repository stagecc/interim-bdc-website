export const allEventsQuery = graphql`
  query {
    events: allMdx(
      sort: {frontmatter: {date: ASC}}
      filter: {
        internal: { contentFilePath: {regex: "/data/events/"} }
      }
    ) {
      edges {
        node {
          frontmatter {
            date(formatString: "MMM D, YYYY")
            display_date
            path
            title
            url
            time
            location
            externalEvent
          }
          excerpt(pruneLength: 280)
        }
      }
    }
  }
`;
